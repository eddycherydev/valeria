Bootstrap File

The public/index.php file is your application's front controller. It boots the core and handles the request lifecycle. 

<?php
require __DIR__ . '/../vendor/autoload.php';

use Core\Router;
use Core\Request;

$router = new Router();
require __DIR__ . '/../config/routes.php';

$response = $router->dispatch(Request::capture());
$response->send();
